
f = @(x) 2^x - 5*x + 2;
a = 0;
b = 1;
tol = 0.001;

if f(a)*f(b) <0
    while abs(b -a) / 2 > tol
        c = (a+b)/ 2;
    
        if f(c) ==0
            break
    
        elseif f(a) * f(c) < 0
            b=c;
        else
            a=c;
        end
    end
    root = (a+b)/2;
    fprintf("final c = %.6f ", root );
    
else
    disp("NO ROOT")

end
