
f = @(x) 2*x^3 -2*x -5;

gx = @(x) ((2*x+5)/2)^(1/3);

a = 1;
b= 2;

x = (a+b)/2;

converge = false;
n = 100;
e = 0.0001

for i = 1:n
    new_x = gx(x);

    if abs(x-new_x) < e
        x = new_x
        converge=true
        break
    end

    x = new_x

end

if converge
    fprintf("converge = %.6f: ", x)
end
