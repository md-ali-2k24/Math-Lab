x = [1891 1901 1911 1921 1931];
y = [46 66 81 93 101];

X = 1925;
n = length(x);

D = zeros(n, n);
D(:,1) = y';

% Backward difference table
for j = 2:n
    for i = n:-1:j
        D(i, j) = D(i, j-1) - D(i-1, j-1);
    end
end

D

h = x(2) - x(1);
p = (X - x(n)) / h;
y = y(n);

for k = 1:n-1
    y = y + p * D(n, k+1);
    p = (p+k)/(k+1) * p;
end

y
