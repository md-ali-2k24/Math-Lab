x= [1891 1901 1911 1921 1931];
y= [46 66 81 93 101];

X = 1895;
n = length(x)

D = zeros(n, n)

D(:, 1) = y'

for j = 2:n
    for i = 1:(n-j+1)
        D(i, j) = D(i+1, j-1) - D(i, j-1);
    end
end
D
h = x(2) - x(1);
p = (X-x(1))/h;
y = y(1);

for k=1:n-1
    y = y + p * D(1, k+1);
    p = (p-k)/(k+1) * p;
end
